import numpy as np
import pretty_midi
import librosa
import soundfile as sf
import subprocess
import os

WAV_FILE = "videoplaybacsk.wav"
MIDI_FILE = "ab.mid"
TEMP_WAV = "temp_output.wav"
OUTPUT_OGG = "output5.ogg"

ROOT_NOTE = 70
SR = 44100

sample, sr = librosa.load(WAV_FILE, sr=SR)

midi = pretty_midi.PrettyMIDI(MIDI_FILE)

end_time = midi.get_end_time()
output = np.zeros(int(end_time * SR) + SR, dtype=np.float32)

for instrument in midi.instruments:
    for note in instrument.notes:
        semitones = note.pitch - ROOT_NOTE
        shifted = librosa.effects.pitch_shift(sample, sr=sr, n_steps=semitones)
        shifted *= (note.velocity / 127.0)
        start = int(note.start * SR)
        end = start + len(shifted)
        if end > len(output):
            output = np.pad(output, (0, end - len(output)))
        output[start:end] += shifted.astype(np.float32)

output /= np.max(np.abs(output) + 1e-6)
TEMP_WAV = "temp.wav" #DO NOT CHANGE
sf.write(TEMP_WAV, output, SR)
subprocess.run([
    "ffmpeg",
    "-y",
    "-i", TEMP_WAV,
    "-c:a", "libvorbis",
    "-q:a", "6",
    OUTPUT_OGG
])

# === DELETE TEMP WAV ===
os.remove(TEMP_WAV)

print("Rendered audio saved to", OUTPUT_OGG)
